97Std_ReturnType CanTp_Transmit(PduIdType TxPduId, const PduInfoType* PduInfoPtr) {
98 Std_ReturnType ret = E_OK;
99 BufReq_ReturnType BufState;
100 PduLengthType availableDataPtr;
101 PduInfoType PduInfoTmp;
102 uint8 SduTable[8];
103 PduInfoTmp.SduDataPtr = SduTable;
104 #ifdef DEBUG
105 printf("\nSduLength: 0x%04x\n",PduInfoPtr->SduLength);
106 #endif
107
108 if(PduInfoPtr->SduDataPtr == NULL) {
109 ret = E_NOT_OK;
110 }
111 else {
112 if(eCanTp_State == CANTP_ON) {
113 if(CanTp_RxTxVariablesConfig.CanTp_TxConfig.eCanTp_TxState == CANTP_TX_WAIT) {
114 if(PduInfoPtr->SduLength < 8) {
115 BufState = PduR_CanTpCopyTxData(TxPduId, &PduInfoTmp, NULL, &availableDataPtr);
116 if(BufState == BUFREQ_OK) {
117 PduInfoTmp.SduDataPtr[0] = CANTP_N_PCI_SF << 4;
118 PduInfoTmp.SduDataPtr[0] = 0x0F & PduInfoPtr->SduLength;
119 for(uint8 i = 0; i < PduInfoPtr->SduLength; i++) {
120 PduInfoTmp.SduDataPtr[i+1] = PduInfoPtr->SduDataPtr[i];
121 }
122 #ifdef DEBUG
123 for(uint8 i = 0; i < 8; i++) {
124 printf("SingleFrame[%d]: 0x%02x\n", i, PduInfoTmp.SduDataPtr[i]);
125 }
126 #endif
127 if(CanIf_Transmit(TxPduId, &PduInfoTmp) == E_OK) {
128 #ifdef DEBUG
129 printf("Timer N_AS start\n");
130 #endif
131 PduR_CanTpTxConfirmation(TxPduId, E_OK);
132 }
133 else {
134 PduR_CanTpTxConfirmation(TxPduId, E_NOT_OK);
135 ret = E_NOT_OK;
136 }
137 }
138 else if(BufState == BUFREQ_E_NOT_OK) {
139 CanTp_RxTxVariablesConfig.CanTp_TxConfig.eCanTp_TxState = CANTP_TX_WAIT;
140 PduR_CanTpTxConfirmation(TxPduId, E_NOT_OK);
141 ret = E_NOT_OK;
142 }
143 else {
144 #ifdef DEBUG
145 printf("Timer N_CS start\n");
146 #endif
147 ret = E_OK;
148 }
149 }
150 else {
151 PduInfoTmp.SduDataPtr[0] = CANTP_N_PCI_FF << 4;
152 if(PduInfoPtr->SduLength <= 4095){
153 PduInfoTmp.SduDataPtr[0] |= (0x0F & (PduInfoPtr->SduLength >> 8));
154 PduInfoTmp.SduDataPtr[1] = (0xFF & (PduInfoPtr->SduLength));
155 PduInfoTmp.SduDataPtr[2] = PduInfoPtr->SduDataPtr[0];
156 PduInfoTmp.SduDataPtr[3] = PduInfoPtr->SduDataPtr[1];
157 PduInfoTmp.SduDataPtr[4] = PduInfoPtr->SduDataPtr[2];
158 PduInfoTmp.SduDataPtr[5] = PduInfoPtr->SduDataPtr[3];
159 PduInfoTmp.SduDataPtr[6] = PduInfoPtr->SduDataPtr[4];
160 PduInfoTmp.SduDataPtr[7] = PduInfoPtr->SduDataPtr[5];
161 }
162 else{
163 PduInfoTmp.SduDataPtr[1] = 0;
164 PduInfoTmp.SduDataPtr[2] = (PduInfoPtr->SduLength >> 24) & 0xFF;
165 PduInfoTmp.SduDataPtr[3] = (PduInfoPtr->SduLength >> 16) & 0xFF;
166 PduInfoTmp.SduDataPtr[4] = (PduInfoPtr->SduLength >> 8) & 0xFF;
167 PduInfoTmp.SduDataPtr[5] = (PduInfoPtr->SduLength >> 0) & 0xFF;
168 PduInfoTmp.SduDataPtr[6] = PduInfoPtr->SduDataPtr[0];
169 PduInfoTmp.SduDataPtr[7] = PduInfoPtr->SduDataPtr[1];
170 }
171 #ifdef DEBUG
172 for(uint8 i = 0; i < 8; i++) {
173 printf("FirstFrame[%d]: 0x%02x\n", i, PduInfoTmp.SduDataPtr[i]);
174 }
175 #endif
176 if(CanIf_Transmit(TxPduId, &PduInfoTmp) == E_OK) {
177 #ifdef DEBUG
178 printf("Timer N_AS start\n");
179 printf("Timer N_BS start\n");
180 #endif
181 PduR_CanTpTxConfirmation(TxPduId, E_OK);
182 CanTp_RxTxVariablesConfig.CanTp_TxConfig.eCanTp_TxState = CANTP_TX_PROCESSING_SUSPEND;
183 }
184 else {
185 PduR_CanTpTxConfirmation(TxPduId, E_NOT_OK);
186 ret = E_NOT_OK;
187 }
188 }
189 }
190 else {
191 ret = E_NOT_OK;
192 }
193 }
194 else {
195 ret = E_NOT_OK;
196 }
197 }
198 return ret;

266 }

269 ret = E_NOT_OK;

311void CanTp_MainFunction(void) {
312 uint16 CalcBlockSize;
313 uint16 SepTime;
314 Std_ReturnType ret;
315 BufReq_ReturnType BufReqState;
316 PduInfoType PduInfoNull = {NULL, NULL, 0};
317 PduInfoType PduInfoPtr;
318 PduLengthType PduLength;
319 TPParameterType parameter1 = TP_BS;
320 TPParameterType parameter2 = TP_STMIN;
321 CanTp_FCFlowStatusType FlowStatus;
322
323 CanTp_ReadParameter(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, TP_STMIN, &SepTime);
324
325 CanTp_IncrementTimer(&CanTp_AsTimer);
326 CanTp_IncrementTimer(&CanTp_BsTimer);
327 CanTp_IncrementTimer(&CanTp_CsTimer);
328 CanTp_IncrementTimer(&CanTp_ArTimer);
329 CanTp_IncrementTimer(&CanTp_BrTimer);
330 CanTp_IncrementTimer(&CanTp_CrTimer);
331
332 if(CanTp_BrTimer.eCanTp_TimerState == START) {
333 BufReqState = PduR_CanTpCopyRxData(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, &PduInfoNull, &PduLength);
334 if(BufReqState == BUFREQ_E_NOT_OK) {
335 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
336 }
337 else {
338 CalcBlockSize = CanTp_CalcBlockSize(PduLength);
339 if(CalcBlockSize > 0) {
340 FlowStatus = FC_CTS;
341 CanTp_ChangeParameter(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, parameter1, CalcBlockSize);
342 CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState = CANTP_RX_PROCESSING;
343 PduInfoPtr.SduDataPtr[0] = CANTP_N_PCI_FC << 4;
344 PduInfoPtr.SduDataPtr[0] |= (0x0F & FlowStatus);
345 PduInfoPtr.SduDataPtr[1] = CalcBlockSize;
346 PduInfoPtr.SduDataPtr[2] = SepTime;
347 ret = CanIf_Transmit(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, &PduInfoPtr);
348 if(ret == E_NOT_OK) {
349 CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState = CANTP_RX_PROCESSING_SUSPEND;
350 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
351 }
352 else {
353 CanTp_ResetTimer(&CanTp_BrTimer);
354 CanTp_ArTimer.eCanTp_TimerState = START;
355 CanTp_CrTimer.eCanTp_TimerState = START;
356 }
357 }
358 if(CanTp_BrTimer.CanTp_Counter >= N_BR_TIMEOUT_VALUE) {
359 CanTp_FCWFTcnt++;
360 CanTp_BrTimer.CanTp_Counter = 0;
361 if(CanTp_FCWFTcnt >= CANTP_FCWFTMAX) {
362 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
363 CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState = CANTP_RX_WAIT;
364 CanTp_FCWFTcnt = 0;
365 }
366 else {
367 FlowStatus = FC_WAIT;
368 PduInfoPtr.SduDataPtr[0] = CANTP_N_PCI_FC << 4;
369 PduInfoPtr.SduDataPtr[0] |= (0x0F & FlowStatus);
370 PduInfoPtr.SduDataPtr[1] = CalcBlockSize;
371 PduInfoPtr.SduDataPtr[2] = SepTime;
372 ret = CanIf_Transmit(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, &PduInfoPtr);
373 if(ret == E_NOT_OK) {
374 CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState = CANTP_RX_PROCESSING_SUSPEND;
375 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
376 }
377 else {
378 CanTp_ArTimer.eCanTp_TimerState = START;
379 CanTp_BrTimer.eCanTp_TimerState = START;
380 }
381 }
382 }
383 }
384 }
385 if(CanTp_CrTimer.eCanTp_TimerState == START) {
386 if(CanTp_CrTimer.CanTp_Counter >= N_CR_TIMEOUT_VALUE) {
387 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
388 CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState = CANTP_RX_PROCESSING_SUSPEND;
389 }
390 }
391 if(CanTp_ArTimer.eCanTp_TimerState == START) {
392 if(CanTp_ArTimer.CanTp_Counter >= N_AR_TIMEOUT_VALUE) {
393 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
394 CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState = CANTP_RX_PROCESSING_SUSPEND;
395 }
396 }
397 if(CanTp_CsTimer.eCanTp_TimerState == START) {
398 if(CanTp_CsTimer.CanTp_Counter >= N_CS_TIMEOUT_VALUE) {
399 PduR_CanTpTxConfirmation(CanTp_RxTxVariablesConfig.CanTp_TxConfig.CanTp_CurrentTxPduId, E_NOT_OK);
400 CanTp_RxTxVariablesConfig.CanTp_TxConfig.eCanTp_TxState = CANTP_TX_PROCESSING_SUSPEND;
401 }
402 }
403 if(CanTp_AsTimer.eCanTp_TimerState == START) {
404 if(CanTp_AsTimer.CanTp_Counter >= N_AS_TIMEOUT_VALUE) {
405 PduR_CanTpTxConfirmation(CanTp_RxTxVariablesConfig.CanTp_TxConfig.CanTp_CurrentTxPduId, E_NOT_OK);
406 CanTp_RxTxVariablesConfig.CanTp_TxConfig.eCanTp_TxState = CANTP_TX_PROCESSING_SUSPEND;
407 }
408 }
409 if(CanTp_BsTimer.eCanTp_TimerState == START) {
410 if(CanTp_BsTimer.CanTp_Counter >= N_BS_TIMEOUT_VALUE) {
411 PduR_CanTpTxConfirmation(CanTp_RxTxVariablesConfig.CanTp_TxConfig.CanTp_CurrentTxPduId, E_NOT_OK);
412 CanTp_RxTxVariablesConfig.CanTp_TxConfig.eCanTp_TxState = CANTP_TX_PROCESSING_SUSPEND;
413 }
414 }

422void CanTp_RxIndication(PduIdType RxPduId, const PduInfoType* PduInfoPtr) {
423 PduInfoType PduInfoTmp;
424 Std_ReturnType ret;
425 uint8 SduTable[8];
426 PduLengthType BuffSize;
427 BufReq_ReturnType BuffState;
428 PduInfoTmp.SduDataPtr = SduTable;
429 CanTp_PCIType CanPCI;
430
431 if(eCanTp_State == CANTP_ON) {
432 switch (PduInfoPtr->SduDataPtr[0] >> 4)
433 {
434 case CANTP_N_PCI_SF:
435 CanPCI.eFrameType = SF;
436 CanPCI.uiFrameLength = PduInfoPtr->SduDataPtr[0];
437 break;
438 case CANTP_N_PCI_FF:
439 CanPCI.eFrameType = FF;
440 if( (PduInfoPtr->SduDataPtr[0] & 0x0F) | PduInfoPtr->SduDataPtr[1] ) {
441 CanPCI.uiFrameLength = PduInfoPtr->SduDataPtr[0] & 0x0F;
442 CanPCI.uiFrameLength = (CanPCI.uiFrameLength << 8) | PduInfoPtr->SduDataPtr[1];
443 }
444 else{
445 CanPCI.uiFrameLength = PduInfoPtr->SduDataPtr[2];
446 CanPCI.uiFrameLength = (CanPCI.uiFrameLength << 8) | PduInfoPtr->SduDataPtr[3];
447 CanPCI.uiFrameLength = (CanPCI.uiFrameLength << 8) | PduInfoPtr->SduDataPtr[4];
448 CanPCI.uiFrameLength = (CanPCI.uiFrameLength << 8) | PduInfoPtr->SduDataPtr[5];
449 }
450 break;
451 case CANTP_N_PCI_FC:
452 CanPCI.eFrameType = FC;
453 CanPCI.uiFlowStatus = PduInfoPtr->SduDataPtr[0] & 0x0F;
454 CanPCI.uiBlockSize = PduInfoPtr->SduDataPtr[1];
455 CanPCI.uiSeparationTime = PduInfoPtr->SduDataPtr[2];
456 break;
457 case CANTP_N_PCI_CF:
458 CanPCI.eFrameType = CF;
459 CanPCI.uiSequenceNumber= PduInfoPtr->SduDataPtr[0] & 0x0F;
460 break;
461 default:
462 CanPCI.eFrameType = 4;
463 break;
464 }
465 switch (CanPCI.eFrameType)
466 {
467 case SF:
468 if(CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState == CANTP_RX_WAIT) {
469 BuffState = PduR_CanTpStartOfReception(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, PduInfoPtr, CanPCI.uiFrameLength, &BuffSize);
470 if(BuffState == BUFREQ_OK) {
471 if(BuffSize >= CanPCI.uiFrameLength) {
472 PduInfoTmp.SduLength = CanPCI.uiFrameLength;
473 PduInfoTmp.SduDataPtr = PduInfoPtr->SduDataPtr+1;
474 BuffState = PduR_CanTpCopyRxData(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, &PduInfoTmp, &BuffSize);
475 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_OK);
476 }
477 else {
478 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
479 }
480 }
481 }
482 else if(CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState == CANTP_RX_PROCESSING) {
483 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
484 CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState = CANTP_RX_WAIT;
485 BuffState = PduR_CanTpStartOfReception(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, PduInfoPtr, CanPCI.uiFrameLength, &BuffSize);
486 if(BuffState == BUFREQ_OK) {
487 if(BuffSize >= CanPCI.uiFrameLength) {
488 PduInfoTmp.SduLength = CanPCI.uiFrameLength;
489 PduInfoTmp.SduDataPtr = PduInfoPtr->SduDataPtr+1;
490 BuffState = PduR_CanTpCopyRxData(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, &PduInfoTmp, &BuffSize);
491 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_OK);
492 }
493 else {
494 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
495 }
496 }
497 }
498 else {
499 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
500 CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState = CANTP_RX_WAIT;
501 BuffState = PduR_CanTpStartOfReception(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, PduInfoPtr, CanPCI.uiFrameLength, &BuffSize);
502 if(BuffState == BUFREQ_OK) {
503 if(BuffSize >= CanPCI.uiFrameLength) {
504 PduInfoTmp.SduLength = CanPCI.uiFrameLength;
505 PduInfoTmp.SduDataPtr = PduInfoPtr->SduDataPtr+1;
506 BuffState = PduR_CanTpCopyRxData(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, &PduInfoTmp, &BuffSize);
507 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_OK);
508 }
509 else {
510 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
511 }
512 }
513 }
514 break;
515 case FF:
516 if(CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState == CANTP_RX_WAIT) {
517 BuffState = PduR_CanTpStartOfReception(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, PduInfoPtr, CanPCI.uiFrameLength, &BuffSize);
518 ret = CanTp_ReceiveFF(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, PduInfoPtr, &CanPCI, BuffState, FC_CTS);
519 }
520 else if(CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState == CANTP_RX_PROCESSING) {
521 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
522 CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState = CANTP_RX_WAIT;
523 ret = CanTp_ReceiveFF(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, PduInfoPtr, &CanPCI, BuffState, FC_WAIT);
524 }
525 else {
526 PduR_CanTpRxIndication(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
527 CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState = CANTP_RX_WAIT;
528 ret = CanTp_ReceiveFF(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, PduInfoPtr, &CanPCI, BuffState, FC_WAIT);
529 }
530 break;
531 case CF:
532 if(CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState == CANTP_RX_PROCESSING) {
533 ret = CanTp_ReceiveCF(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, PduInfoPtr, &CanPCI, FC_CTS);
534 }
535 else {
536 PduR_CanTpRxIndication (CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, E_NOT_OK);
537 CanTp_RxTxVariablesConfig.CanTp_RxConfig.eCanTp_RxState = CANTP_RX_WAIT;
538 }
539 break;
540 case FC:
541 ret = CanTp_ReceiveFC(CanTp_RxTxVariablesConfig.CanTp_RxConfig.CanTp_CurrentRxPduId, PduInfoPtr, &CanPCI, FC_CTS);
542 break;
543 default:
544 #ifdef DEBUG
545 printf("\nERROR 404 :)\n");
546 #endif
547 break;
548 }
549 }

556void CanTp_TxConfirmation(PduIdType TxPduId, Std_ReturnType result) {
